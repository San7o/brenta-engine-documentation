320 template <typename C, typename... Components, typename N = none>
321 static std::vector<entity_t> query_components()
322 {
323 if (!world::components)
324 {
325 OAK_ERROR("Cannot query: world not initialized");
326 return {};
327 }
328
329 std::vector<entity_t> matched;
330
331 if (!components->count(std::type_index(typeid(C))))
332 {
333 return matched;
334 }
335
336 for (auto component : components->at(std::type_index(typeid(C))))
337 {
338 matched.push_back(component->entity);
339 }
340
341 if (matched.empty())
342 return matched;
343 if (sizeof...(Components) == 0)
344 return matched;
345
346 query_components_rec<Components..., none>(&matched);
347
348 return matched;
349 }
350
351 template <typename C, typename... Components>
352 static void query_components_rec(std::vector<entity_t> *entities)
353 {
354 if (entities->empty())
355 return;
356 std::vector<entity_t> matched;
357
358 if (!components->count(std::type_index(typeid(C))))
359 {
360 *entities = matched;
361 return;
362 }
363 for (auto component : components->at(std::type_index(typeid(C))))
364 {
365 if (std::find(entities->begin(), entities->end(),
366 (int &) component->entity)
367 != entities->end())
368 {
369 matched.push_back(component->entity);
370 }
371 }
372 *entities = matched;
373
374 if (sizeof...(Components) == 0)
375 return;
376
377 (query_components_rec<Components>(entities), ...);
378 }
