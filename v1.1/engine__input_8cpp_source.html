44void input::add_keyboard_callback(int key, std::function<void()> callback)
45{
46 input::keyboard_callbacks[key] = callback;
47 INFO("Added callback for key: {}", std::to_string(key));
48}

50void input::remove_keyboard_callback(int key)
51{
52 if (input::keyboard_callbacks.find(key) == input::keyboard_callbacks.end())
53 {
54 ERROR("No callback found for key: {}", key);
55 return;
56 }
57
58 input::keyboard_callbacks.erase(key);
59 INFO("Removed callback for key: {}", key);
60}

62void input::key_callback(GLFWwindow *window, int key, int scancode, int action,
63 int mods)
64{
65 if (action == GLFW_PRESS)
66 {
67 if (input::keyboard_callbacks.find(key)
68 != input::keyboard_callbacks.end())
69 {
70 input::keyboard_callbacks.at(key)();
71 }
72 }
73}

95void input::mouse_pos_callback(GLFWwindow *window, double xpos, double ypos)
